from __future__ import annotations

from dataclasses import dataclass

import numpy as np
import pandas as pd


@dataclass(frozen=True)
class EdasResult:
    average_solution: pd.Series
    pda: pd.DataFrame
    nda: pd.DataFrame
    sp: pd.Series
    sn: pd.Series
    nsp: pd.Series
    nsn: pd.Series
    appraisal_score: pd.Series
    ranking: pd.Series


def _safe_divide(numerator: pd.DataFrame | pd.Series, denominator: pd.Series | float) -> pd.DataFrame | pd.Series:
    if isinstance(denominator, (int, float)):
        if denominator == 0:
            return numerator * 0
        return numerator / denominator

    safe_denominator = denominator.replace(0, np.nan)
    result = numerator.div(safe_denominator, axis=1 if isinstance(numerator, pd.DataFrame) else 0)
    return result.fillna(0.0)


def run_edas(
    decision_matrix: pd.DataFrame,
    weights: dict[str, float],
    criteria_types: dict[str, str],
) -> EdasResult:
    if decision_matrix.empty:
        raise ValueError("Decision matrix kosong.")

    criteria = list(decision_matrix.columns)

    missing_weights = [column for column in criteria if column not in weights]
    if missing_weights:
        raise ValueError(f"Bobot untuk kriteria berikut belum tersedia: {missing_weights}")

    missing_types = [column for column in criteria if column not in criteria_types]
    if missing_types:
        raise ValueError(f"Jenis kriteria untuk kolom berikut belum tersedia: {missing_types}")

    weights_series = pd.Series({column: float(weights[column]) for column in criteria})
    total_weight = float(weights_series.sum())
    if total_weight <= 0:
        raise ValueError("Total bobot harus lebih dari 0.")

    weights_series = weights_series / total_weight

    matrix = decision_matrix[criteria].astype(float)
    average_solution = matrix.mean(axis=0)

    pda = pd.DataFrame(0.0, index=matrix.index, columns=criteria)
    nda = pd.DataFrame(0.0, index=matrix.index, columns=criteria)

    for criterion in criteria:
        av = average_solution[criterion]
        values = matrix[criterion]
        criterion_type = str(criteria_types[criterion]).lower().strip()

        if criterion_type == "cost":
            pda_column = (av - values).clip(lower=0)
            nda_column = (values - av).clip(lower=0)
        else:
            pda_column = (values - av).clip(lower=0)
            nda_column = (av - values).clip(lower=0)

        if av == 0:
            pda[criterion] = 0.0
            nda[criterion] = 0.0
        else:
            pda[criterion] = pda_column / av
            nda[criterion] = nda_column / av

    weighted_pda = pda.mul(weights_series, axis=1)
    weighted_nda = nda.mul(weights_series, axis=1)

    sp = weighted_pda.sum(axis=1)
    sn = weighted_nda.sum(axis=1)

    max_sp = float(sp.max())
    max_sn = float(sn.max())

    if max_sp == 0:
        nsp = pd.Series(0.0, index=sp.index)
    else:
        nsp = sp / max_sp

    if max_sn == 0:
        nsn = pd.Series(1.0, index=sn.index)
    else:
        nsn = 1 - (sn / max_sn)

    appraisal_score = 0.5 * (nsp + nsn)
    ranking = appraisal_score.rank(ascending=False, method="min").astype(int)

    return EdasResult(
        average_solution=average_solution,
        pda=pda,
        nda=nda,
        sp=sp,
        sn=sn,
        nsp=nsp,
        nsn=nsn,
        appraisal_score=appraisal_score,
        ranking=ranking,
    )

from __future__ import annotations

from dataclasses import dataclass

import pandas as pd


@dataclass(frozen=True)
class PlaylistBuildResult:
    playlist_df: pd.DataFrame
    total_duration_seconds: int
    target_duration_seconds: int
    remaining_seconds: int
    exceeded_seconds: int


def build_playlist_by_duration(
    ranked_songs: pd.DataFrame,
    target_duration_minutes: int,
) -> PlaylistBuildResult:
    if target_duration_minutes <= 0:
        empty_df = pd.DataFrame(columns=list(ranked_songs.columns) + ["playlist_order", "cumulative_duration_seconds"])
        return PlaylistBuildResult(
            playlist_df=empty_df,
            total_duration_seconds=0,
            target_duration_seconds=0,
            remaining_seconds=0,
            exceeded_seconds=0,
        )

    required_columns = {"duration_seconds", "appraisal_score", "rank"}
    missing_columns = sorted(required_columns - set(ranked_songs.columns))
    if missing_columns:
        raise ValueError(f"Kolom wajib untuk pembentukan playlist belum tersedia: {missing_columns}")

    target_duration_seconds = int(target_duration_minutes * 60)

    ordered_df = ranked_songs.sort_values(
        by=["appraisal_score", "rank"],
        ascending=[False, True],
    ).reset_index(drop=True)

    selected_rows: list[dict] = []
    total_duration_seconds = 0

    for _, row in ordered_df.iterrows():
        row_duration = int(row["duration_seconds"])
        row_dict = row.to_dict()

        total_duration_seconds += row_duration
        row_dict["playlist_order"] = len(selected_rows) + 1
        row_dict["cumulative_duration_seconds"] = total_duration_seconds
        selected_rows.append(row_dict)

        if total_duration_seconds >= target_duration_seconds:
            break

    playlist_df = pd.DataFrame(selected_rows)

    remaining_seconds = max(0, target_duration_seconds - total_duration_seconds)
    exceeded_seconds = max(0, total_duration_seconds - target_duration_seconds)

    return PlaylistBuildResult(
        playlist_df=playlist_df,
        total_duration_seconds=total_duration_seconds,
        target_duration_seconds=target_duration_seconds,
        remaining_seconds=remaining_seconds,
        exceeded_seconds=exceeded_seconds,
    )

PARAMETERS = [
    "bpm",
    "energy",
    "danceability",
    "happiness",
    "popularity",
    "acousticness",
    "instrumentalness",

@dataclass(frozen=True)
class QuestionnaireItem:
    question_id: int
    parameter: str
    positive_toward_higher_value: bool


@dataclass(frozen=True)
class PreferenceResult:
    parameter_scores: dict[str, float]
    weights: dict[str, float]
    criteria_types: dict[str, str]


QUESTIONNAIRE_MAPPING = [
    QuestionnaireItem(question_id=1, parameter="bpm", positive_toward_higher_value=True),
    QuestionnaireItem(question_id=2, parameter="bpm", positive_toward_higher_value=False),
    QuestionnaireItem(question_id=3, parameter="energy", positive_toward_higher_value=True),
    QuestionnaireItem(question_id=4, parameter="energy", positive_toward_higher_value=False),
    QuestionnaireItem(question_id=5, parameter="energy", positive_toward_higher_value=True),
    QuestionnaireItem(question_id=6, parameter="danceability", positive_toward_higher_value=True),
    QuestionnaireItem(question_id=7, parameter="danceability", positive_toward_higher_value=True),
    QuestionnaireItem(question_id=8, parameter="happiness", positive_toward_higher_value=True),
    QuestionnaireItem(question_id=9, parameter="popularity", positive_toward_higher_value=True),
    QuestionnaireItem(question_id=10, parameter="acousticness", positive_toward_higher_value=True),
    QuestionnaireItem(question_id=11, parameter="instrumentalness", positive_toward_higher_value=True),
    QuestionnaireItem(question_id=12, parameter="instrumentalness", positive_toward_higher_value=True),
    QuestionnaireItem(question_id=13, parameter="speechiness", positive_toward_higher_value=True),
    QuestionnaireItem(question_id=14, parameter="speechiness", positive_toward_higher_value=False),
]


def _validate_answers(answers: dict[int, int]) -> None:
    expected_ids = {item.question_id for item in QUESTIONNAIRE_MAPPING}
    given_ids = set(answers.keys())

    missing = sorted(expected_ids - given_ids)
    extra = sorted(given_ids - expected_ids)

    if missing:
        raise ValueError(f"Jawaban belum lengkap. Nomor yang belum diisi: {missing}")

    if extra:
        raise ValueError(f"Terdapat nomor pertanyaan tidak valid: {extra}")

    for question_id, answer in answers.items():
        if answer < 1 or answer > 5:
            raise ValueError(f"Skala Likert harus 1-5. Nilai invalid pada nomor {question_id}: {answer}")


def _normalize_answer(answer: int, positive_toward_higher_value: bool) -> float:
    if positive_toward_higher_value:
        return float(answer)

    return float(6 - answer)


def _classify_criterion(parameter_score: float) -> str:
    if parameter_score > 3.0:
        return "benefit"
    if parameter_score < 3.0:
        return "cost"
    return "neutral"


def build_preferences_from_questionnaire(answers: dict[int, int]) -> PreferenceResult:
    _validate_answers(answers)

    score_bucket: dict[str, list[float]] = {parameter: [] for parameter in PARAMETERS}

    for item in QUESTIONNAIRE_MAPPING:
        raw_answer = answers[item.question_id]
        normalized_answer = _normalize_answer(raw_answer, item.positive_toward_higher_value)
        score_bucket[item.parameter].append(normalized_answer)

    parameter_scores: dict[str, float] = {}
    for parameter in PARAMETERS:
        parameter_values = score_bucket[parameter]
        parameter_scores[parameter] = sum(parameter_values) / len(parameter_values)

    total_score = sum(parameter_scores.values())
    if total_score <= 0:
        raise ValueError("Total skor preferensi tidak valid untuk normalisasi bobot.")

    weights = {
        parameter: score / total_score
        for parameter, score in parameter_scores.items()
    }

    criteria_types = {
        parameter: _classify_criterion(score)
        for parameter, score in parameter_scores.items()
    }

    return PreferenceResult(
        parameter_scores=parameter_scores,
        weights=weights,
        criteria_types=criteria_types,
    )

from __future__ import annotations

from dataclasses import dataclass
from typing import Iterable

import pandas as pd


EDAS_CRITERIA = [
    "bpm",
    "energy",
    "danceability",
    "happiness",
    "popularity",
    "acousticness",
    "instrumentalness",
    "speechiness",
]


@dataclass(frozen=True)
class PreprocessResult:
    cleaned_df: pd.DataFrame
    decision_matrix: pd.DataFrame


def parse_duration_to_seconds(duration_text: str) -> int:
    value = str(duration_text).strip()
    parts = value.split(":")

    if len(parts) != 2:
        raise ValueError(f"Format durasi tidak valid: {duration_text}")

    minutes_text, seconds_text = parts
    minutes = int(minutes_text)
    seconds = int(seconds_text)

    if minutes < 0 or seconds < 0:
        raise ValueError(f"Nilai durasi tidak valid: {duration_text}")

    normalized_minutes = minutes + (seconds // 60)
    normalized_seconds = seconds % 60

    return normalized_minutes * 60 + normalized_seconds


def _safe_parse_duration_to_seconds(duration_text: str) -> int | None:
    try:
        return parse_duration_to_seconds(duration_text)
    except (TypeError, ValueError):
        return None


def _ensure_columns(df: pd.DataFrame, required_columns: Iterable[str]) -> None:
    missing = [column for column in required_columns if column not in df.columns]
    if missing:
        raise KeyError(f"Kolom wajib tidak ditemukan: {missing}")


def preprocess_for_edas(df: pd.DataFrame) -> PreprocessResult:
    required_columns = ["artist", "title", "duration", "status", *EDAS_CRITERIA]
    _ensure_columns(df, required_columns)

    working_df = df.copy()

    if "status" in working_df.columns:
        working_df["status"] = working_df["status"].astype(str).str.strip().str.lower()
        working_df = working_df[working_df["status"] == "ok"].copy()

    working_df["duration_seconds"] = working_df["duration"].apply(_safe_parse_duration_to_seconds)

    for column in EDAS_CRITERIA:
        working_df[column] = pd.to_numeric(working_df[column], errors="coerce")

    working_df = working_df.dropna(subset=[*EDAS_CRITERIA, "duration_seconds"]).copy()
    working_df["duration_seconds"] = working_df["duration_seconds"].astype(int)

    decision_matrix = working_df[EDAS_CRITERIA].copy()

    if decision_matrix.empty:
        raise ValueError("Decision matrix kosong setelah preprocessing.")

    return PreprocessResult(cleaned_df=working_df, decision_matrix=decision_matrix)

import pandas as pd

from src.edas import run_edas


class TestEdas(unittest.TestCase):
    def test_edas_expected_ranking_on_small_matrix(self) -> None:
        decision_matrix = pd.DataFrame(
            {
                "c1": [10.0, 20.0, 15.0],
                "c2": [20.0, 10.0, 15.0],
            }
        )

        weights = {"c1": 0.5, "c2": 0.5}
        criteria_types = {"c1": "benefit", "c2": "cost"}

        result = run_edas(decision_matrix, weights, criteria_types)

        self.assertEqual(int(result.ranking.iloc[1]), 1)
        self.assertEqual(int(result.ranking.iloc[2]), 2)
        self.assertEqual(int(result.ranking.iloc[0]), 3)

TRACEABILITY RUMUS EDAS
Paper -> Implementasi Kode

Dokumen ini dibuat untuk membuktikan bahwa rumus-rumus EDAS pada paper benar-benar
terpakai di implementasi kode, termasuk lokasi file dan cuplikan kodenya.

====================================================================
A. RUMUS BOBOT KRITERIA (dari preferensi kuesioner)
====================================================================
Rumus di paper:
wj = Sj / sum(Sj)

Makna:
- S_j = skor preferensi parameter ke-j (hasil agregasi jawaban kuesioner)
- w_j = bobot kriteria ke-j

Implementasi kode:
- File: backend/src/preferences.py
- Fungsi: build_preferences_from_questionnaire(...)

Cuplikan kode:
weights = {
    parameter: score / total_score
    for parameter, score in parameter_scores.items()
}

Catatan:
- Ini identik dengan normalisasi bobot di paper.

====================================================================
B. PENENTUAN JENIS KRITERIA (benefit/cost/netral)
====================================================================
Aturan di paper:
- skor > 3  -> benefit
- skor < 3  -> cost
- skor = 3  -> netral (moderate, tidak dibalik ekstrem)

Implementasi kode:
- File: backend/src/preferences.py
- Fungsi: _classify_criterion(...)

Cuplikan kode:
def _classify_criterion(parameter_score: float) -> str:
    if parameter_score > 3.0:
        return "benefit"
    if parameter_score < 3.0:
        return "cost"
    return "neutral"

Catatan:
- Kriteria "neutral" diperlakukan tanpa pembalikan arah (diproses seperti benefit)
  agar tetap moderat, sesuai narasi paper.

====================================================================
C. RUMUS AV (Average Solution)
====================================================================
Rumus di paper:
AV_j = (1/m) * sum_i(x_ij)

Implementasi kode:
- File: backend/src/edas.py
- Fungsi: run_edas(...)

Cuplikan kode:
average_solution = matrix.mean(axis=0)

Catatan:
- mean(axis=0) menghitung rata-rata per kolom kriteria = AV_j.

====================================================================
D. RUMUS PDA/NDA UNTUK BENEFIT DAN COST
====================================================================
Rumus di paper:
Untuk benefit:
PDA_ij = max(0, x_ij - AV_j) / AV_j
NDA_ij = max(0, AV_j - x_ij) / AV_j

Untuk cost:
PDA_ij = max(0, AV_j - x_ij) / AV_j
NDA_ij = max(0, x_ij - AV_j) / AV_j

Implementasi kode:
- File: backend/src/edas.py
- Fungsi: run_edas(...)

Cuplikan kode:
if criterion_type == "cost":
    pda_column = (av - values).clip(lower=0)
    nda_column = (values - av).clip(lower=0)
else:
    pda_column = (values - av).clip(lower=0)
    nda_column = (av - values).clip(lower=0)

if av == 0:
    pda[criterion] = 0.0
    nda[criterion] = 0.0
else:
    pda[criterion] = pda_column / av
    nda[criterion] = nda_column / av

Catatan:
- clip(lower=0) setara dengan operasi max(0, ...).
- Penanganan av == 0 adalah safeguard numerik (agar tidak division by zero).

====================================================================
E. RUMUS SP DAN SN (penjumlahan berbobot)
====================================================================
Rumus di paper:
SP_i = sum_j (w_j * PDA_ij)
SN_i = sum_j (w_j * NDA_ij)

Implementasi kode:
- File: backend/src/edas.py
- Fungsi: run_edas(...)

Cuplikan kode:
weighted_pda = pda.mul(weights_series, axis=1)
weighted_nda = nda.mul(weights_series, axis=1)

sp = weighted_pda.sum(axis=1)
sn = weighted_nda.sum(axis=1)

Catatan:
- multiply per kolom dengan bobot lalu sum per baris = SP_i dan SN_i.

====================================================================
F. RUMUS NSP DAN NSN (normalisasi)
====================================================================
Rumus di paper:
NSP_i = SP_i / max(SP_i)
NSN_i = 1 - (SN_i / max(SN_i))

Implementasi kode:
- File: backend/src/edas.py
- Fungsi: run_edas(...)

Cuplikan kode:
max_sp = float(sp.max())
max_sn = float(sn.max())

if max_sp == 0:
    nsp = pd.Series(0.0, index=sp.index)
else:
    nsp = sp / max_sp

if max_sn == 0:
    nsn = pd.Series(1.0, index=sn.index)
else:
    nsn = 1 - (sn / max_sn)

Catatan:
- Bentuk normalisasi sama dengan paper.
- Kondisi max=0 adalah safeguard numerik.

====================================================================
G. RUMUS APPRAISAL SCORE (AS)
====================================================================
Rumus di paper:
AS_i = (NSP_i + NSN_i) / 2

Implementasi kode:
- File: backend/src/edas.py
- Fungsi: run_edas(...)

Catatan:

1.1 Latar Belakang
Musik telah menjadi bagian yang melekat dalam kehidupan sehari-hari manusia dan hadir dalam berbagai situasi aktivitas harian. Dalam praktik kehidupan nyata, individu berinteraksi dengan musik tidak hanya sebagai hiburan utama, tetapi juga sebagai elemen yang menyertai berbagai aktivitas, seperti bekerja, belajar, berolahraga, maupun beristirahat. Pola penggunaan ini menunjukkan bahwa musik digunakan secara kontekstual sebagai bagian dari rutinitas hidup, sehingga pemilihan musik sering kali disesuaikan dengan situasi aktivitas saat musik digunakan.
Penggunaan musik dalam kehidupan sehari-hari tidak bersifat acak, melainkan dipengaruhi oleh konteks situasional yang menyertai aktivitas individu. Penelitian berbasis pengamatan situasi nyata menunjukkan bahwa sesi mendengarkan musik yang terjadi pada konteks berbeda memiliki perbedaan signifikan dalam perilaku mendengarkan, seperti tingkat fokus terhadap musik, kecenderungan melewati lagu, serta variasi genre yang dipilih. Temuan ini menegaskan bahwa konteks aktivitas berperan penting dalam membentuk pola interaksi pengguna dengan musik dalam kehidupan sehari-hari (Hausberger et al., 2025).
Perkembangan layanan streaming musik memberikan pengguna akses terhadap koleksi lagu dalam jumlah yang sangat besar dalam satu platform digital. Meskipun kondisi ini memperluas pilihan pengguna, ketersediaan opsi dalam jumlah masif tidak selalu mempermudah proses pemilihan. Pengguna dapat mengalami kondisi choice overload, yaitu situasi ketika banyaknya alternatif justru meningkatkan beban kognitif dan membuat proses memilih musik terasa lebih sulit serta kurang menyenangkan. Dalam konteks pemilihan musik, choice overload berpotensi menurunkan kenyamanan pengguna dan berdampak pada kualitas pengalaman mendengarkan musik secara keseluruhan (Schulz, 2021). Kondisi tersebut menunjukkan bahwa ketersediaan koleksi musik yang sangat besar tidak selalu meningkatkan kemudahan dalam memilih musik, melainkan justru dapat menciptakan kebutuhan terhadap mekanisme yang mampu membantu pengguna menentukan pilihan musik yang paling sesuai dengan kebutuhannya.
Untuk membantu pengguna menghadapi permasalahan tersebut, sistem rekomendasi musik menjadi komponen penting dalam layanan streaming digital. Namun, rekomendasi musik memiliki karakteristik yang berbeda dibandingkan domain rekomendasi lain. Musik umumnya dikonsumsi dalam durasi singkat, sering didengarkan secara berurutan dalam bentuk playlist, serta digunakan dalam berbagai konteks aktivitas yang berbeda. Selain itu, preferensi musik tidak selalu bersifat statis, melainkan dapat berubah mengikuti kondisi situasional pengguna. Oleh karena itu, pendekatan rekomendasi yang hanya mengandalkan histori mendengarkan atau kemiripan lagu yang umum digunakan pada layanan streaming musik modern sering kali belum cukup untuk merepresentasikan kebutuhan penggunaan musik secara kontekstual (Zeng et al., 2024).
Dalam konteks pemodelan komputasional, karakteristik lagu dapat direpresentasikan melalui parameter audio yang bersifat kuantitatif. Analisis berbasis data berskala besar pada platform streaming musik menunjukkan bahwa musik yang digunakan pada konteks aktivitas tertentu, seperti belajar dan tidur, memiliki pola karakteristik audio yang dapat diamati melalui fitur audio, seperti tempo, energy, danceability, happiness, dan fitur lainnya. Temuan ini menunjukkan bahwa perbedaan konteks penggunaan musik dapat dimodelkan melalui representasi karakteristik lagu berbasis parameter audio, tanpa perlu mengaitkannya secara langsung dengan klaim efek psikologis tertentu (Scarratt et al., 2023).
Meskipun berbagai sistem rekomendasi musik telah dikembangkan, sebagian besar pendekatan masih berfokus pada kemiripan lagu atau histori mendengarkan pengguna, tanpa secara eksplisit menilai kesesuaian lagu terhadap konteks aktivitas saat musik digunakan. Padahal, perubahan konteks aktivitas diketahui memengaruhi pola penggunaan musik (Hausberger et al., 2025), sehingga diperlukan pendekatan yang mampu mempertimbangkan beberapa parameter lagu secara bersamaan untuk memodelkan tingkat kesesuaian lagu dengan konteks penggunaan tertentu (Zeng et al., 2024).
Berdasarkan kebutuhan tersebut, penelitian ini mengembangkan sistem rekomendasi playlist musik berbasis konteks aktivitas pengguna dengan memanfaatkan parameter audio lagu sebagai kriteria evaluasi yang pembobotannya ditentukan berdasarkan preferensi pengguna, metode Evaluation Based on Distance from Average Solution (EDAS) sebagai mekanisme penilaian multikriteria, serta Natural Language Generation (NLG) untuk menyajikan hasil rekomendasi dalam bentuk penjelasan yang mudah dipahami. Pendekatan ini diharapkan dapat membantu pengguna memperoleh rekomendasi playlist yang lebih sesuai dengan konteks aktivitasnya, sekaligus memberikan transparansi terhadap proses penilaian yang dilakukan oleh sistem.

2.1 Sistem Rekomendasi Musik
Sistem rekomendasi merupakan instrumen krusial dalam ekosistem layanan musik digital untuk membantu pengguna menyeleksi lagu yang relevan di tengah melimpahnya pilihan. Seiring dengan pesatnya perkembangan platform seperti Spotify dan Apple Music, kebutuhan akan personalisasi yang mampu beradaptasi dengan pola konsumsi musik pengguna secara dinamis menjadi semakin tinggi. Menurut Zeng et al. (2024), evolusi teknologi ini bergerak dari metode konvensional seperti content-based filtering dan collaborative filtering menuju model pembelajaran mendalam (deep learning) yang lebih canggih. Dalam penerapannya, banyak sistem modern kini mengadopsi pendekatan hybrid yang menggabungkan analisis karakteristik lagu dengan pola interaksi antar pengguna demi mengoptimalkan akurasi serta relevansi saran yang diberikan.
Lebih lanjut, Zeng et al. (2024) menjelaskan bahwa domain musik menghadapi tantangan unik berupa subjektivitas preferensi yang tinggi, perubahan selera yang cepat, serta kendala cold-start bagi pengguna dengan data historis terbatas. Kondisi ini mendorong riset untuk terus mengintegrasikan berbagai sumber data, mulai dari fitur audio dan metadata hingga konteks penggunaan, guna meningkatkan kualitas rekomendasi. Di samping aspek algoritma, efektivitas sistem juga bergantung pada evaluasi yang ketat menggunakan metrik seperti precision, recall, dan normalized discounted cumulative gain (NDCG) untuk menjamin hasil yang diberikan selaras dengan preferensi nyata pengguna. Dengan demikian, konsep sistem rekomendasi dalam penelitian ini diposisikan sebagai fondasi dalam merancang mekanisme saran playlist berbasis konteks yang didukung oleh kerangka evaluasi terstruktur.
2.2 Context-Aware Music Recommender
Evolusi dalam mekanisme rekomendasi musik saat ini menandai adanya transisi dari metode tradisional menuju model yang lebih responsif terhadap situasi serta aspek afektif pengguna. Berdasarkan studi literatur sistematis terhadap 64 artikel ilmiah, Assun  o et al. (2022) mengidentifikasi bahwa kecenderungan seseorang dalam mendengarkan musik tidak sekadar ditentukan oleh riwayat interaksi atau kemiripan konten semata. Faktor eksternal seperti jenis aktivitas, waktu, lokasi, hingga kondisi emosional pengguna dilaporkan berpengaruh dalam berbagai penelitian. Dalam kerangka ini, context-aware recommender system diartikan sebagai teknologi yang mengintegrasikan informasi situasional-seperti rutinitas harian, momen penggunaan, atau kondisi lingkungan fisik-untuk memperdalam akurasi proses pemberian saran.
Lebih mendalam, Assun  o et al. (2022) memaparkan bahwa konteks dalam sistem komputasi mencakup seluruh data yang mampu mendeskripsikan kondisi suatu entitas, baik itu pengguna maupun aplikasi yang digunakan. Pada ranah musik, variabel konteks ini dapat berupa aktivitas spesifik (seperti belajar atau berolahraga), pembagian waktu (pagi hingga malam), cuaca, hingga keterlibatan sosial. Penyematan data kontekstual ini krusial untuk meningkatkan relevansi, mengingat kebutuhan musik individu bersifat sangat dinamis dan bergantung pada situasi yang sedang dialami.
Di samping faktor situasi, dimensi emosi juga menjadi fokus utama dalam inovasi music recommender. Berbagai literatur menunjukkan bahwa musik berperan sebagai instrumen regulasi emosi, baik untuk memicu perasaan tertentu maupun menjaga suasana hati yang ada. Dalam arsitektur sistem rekomendasi, emosi ditinjau dari dua sudut pandang: perasaan yang dialami pendengar dan impresi emosional yang terpancar dari karya musik itu sendiri. Representasi emosi ini biasanya dipetakan melalui model kategorikal (seperti rasa senang atau sedih) atau model dimensional valence-arousal yang menempatkan emosi dalam koordinat ruang kontinu.
Tinjauan tersebut juga menegaskan bahwa sinergi antara aspek konteks dan emosi secara simultan mampu menghadirkan pengalaman mendengarkan yang jauh lebih personal dibandingkan metode yang hanya mengandalkan satu aspek. Integrasi ini memungkinkan platform tidak hanya mengenali selera pengguna, tetapi juga memahami kondisi situasional dan emosional yang paling tepat untuk menyarankan sebuah lagu. Oleh karena itu, sistem yang berbasis context-aware dan emotion-aware dipandang sebagai pengembangan lebih lanjut dari model rekomendasi konvensional yang terpaku pada data historis semata.
Berlandaskan pada teori tersebut, penelitian ini mengadopsi konsep context-aware recommendation dengan memfokuskan variabel pada aktivitas utama, waktu, durasi, serta preferensi situasional yang diberikan pengguna ke dalam sistem. Strategi ini tidak ditujukan untuk melakukan pengukuran psikologis secara mendalam, melainkan memfungsikan data konteks tersebut sebagai parameter kunci dalam mekanisme evaluasi serta pemeringkatan lagu yang dilakukan secara sistematis.
2.3 Representasi Lagu Berbasis Parameter Audio
Dalam arsitektur sistem rekomendasi kontemporer, penggambaran sebuah karya musik telah melampaui batasan metadata konvensional seperti judul, musisi, atau klasifikasi genre. Saat ini, platform streaming memanfaatkan ekstraksi fitur audio kuantitatif secara otomatis untuk memodelkan lagu sebagai entitas dengan atribut numerik yang terukur. Melalui parameter komputasional ini, proses analisis dan komparasi antar lagu dapat dilaksanakan secara lebih sistematis dan objektif dalam kerangka kerja rekomendasi.
Menurut Scarratt et al. (2023), fitur audio yang disediakan oleh platform seperti Spotify sangat efektif untuk mengenali pola kemiripan dalam kumpulan data musik berskala besar. Melalui observasi terhadap berbagai daftar putar (playlist), studi tersebut mengungkapkan bahwa distribusi variabel seperti energy, valence, tempo, dan loudness menunjukkan konsistensi yang kuat pada kategori musik tertentu. Temuan ini memperkuat posisi parameter audio sebagai representasi komputasional yang valid, yang mampu mendefinisikan karakteristik lagu tanpa harus terpaku pada interpretasi subjektif manusia.
Penelitian ini secara spesifik mengintegrasikan sejumlah fitur audio utama, yang meliputi tempo (BPM), energy, danceability, valence, acousticness, instrumentalness, speechiness, dan popularity. Secara teknis, tempo mengukur laju ketukan lagu, sementara energy merefleksikan intensitas serta dinamika musikalitasnya. Danceability menilai aspek ritmik dan stabilitas ketukan untuk aktivitas fisik, sedangkan valence memetakan nuansa emosional lagu dari spektrum negatif ke positif. Di sisi lain, acousticness menunjukkan proporsi instrumen non-elektronik, instrumentalness memprediksi ketiadaan elemen vokal, speechiness mendeteksi densitas kata-kata lisan, dan popularity menjadi indikator tingkat penerimaan publik di platform digital.
Penggunaan parameter-parameter tersebut memungkinkan transformasi setiap lagu ke dalam bentuk vektor numerik, sehingga penilaian kesesuaian dapat dilakukan secara presisi. Dalam kerangka penelitian ini, fitur-fitur audio tersebut difungsikan sebagai kriteria utama dalam mekanisme pengambilan keputusan multikriteria. Dengan demikian, setiap kandidat lagu dapat dianalisis dan ditentukan urutan peringkatnya secara terstruktur sesuai dengan preferensi yang telah ditetapkan pengguna.
2.4 Multi-Criteria Decision Making (MCDM)
Dalam sistem rekomendasi yang mengandalkan representasi numerik melalui berbagai parameter audio, proses penentuan alternatif terbaik tidak lagi bertumpu pada satu atribut tunggal, melainkan melibatkan pertimbangan sejumlah kriteria secara simultan. Penggunaan fitur seperti tempo, energy, valence, dan danceability menuntut adanya pendekatan pengambilan keputusan yang mampu menyatukan berbagai dimensi tersebut secara sistematis. Kerangka kerja yang sangat relevan untuk mengatasi kompleksitas ini adalah Multi-Criteria Decision Making atau MCDM.
Pengambilan keputusan multikriteria merupakan sebuah metodologi yang bertujuan untuk menetapkan alternatif paling optimal dengan melibatkan lebih dari satu kriteria evaluasi (Taherdoost & Madanchian, 2023). Secara fundamental, struktur permasalahan dalam MCDM dibangun dari himpunan alternatif, kriteria, serta bobot yang merefleksikan tingkat kepentingan masing-masing variabel. Data-data ini diorganisir dalam sebuah matriks keputusan, di mana baris merepresentasikan alternatif lagu dan kolom menyatakan kriteria audio, sementara nilai di setiap sel menunjukkan performa relatif lagu terhadap kriteria tersebut. Dengan bantuan vektor bobot, proses agregasi dilakukan untuk memperoleh skor komprehensif yang menjadi basis pemeringkatan akhir.
Secara kategoris, MCDM terbagi menjadi dua kelompok besar, yakni Multi-Attribute Decision Making (MADM) dan Multi-Objective Decision Making (MODM) (Taherdoost & Madanchian, 2023). MADM berfokus pada penilaian terhadap sejumlah alternatif yang jumlahnya terbatas dan telah terdefinisi secara jelas, sedangkan MODM lebih diarahkan pada optimasi dalam ruang solusi yang bersifat kontinu. Mengingat sistem ini melakukan evaluasi terhadap sekumpulan lagu yang sudah tersedia guna menentukan prioritas berdasarkan keinginan pengguna, maka permasalahan dalam penelitian ini diklasifikasikan ke dalam kategori MADM.
Terdapat beragam algoritma yang dikembangkan dalam kerangka MCDM, termasuk AHP, TOPSIS, VIKOR, hingga EDAS, yang masing-masing menawarkan logika perhitungan dan karakteristik yang berbeda. Studi komparatif mengungkapkan bahwa pemilihan metode yang tidak tepat dapat menghasilkan urutan peringkat yang berbeda meskipun diterapkan pada data yang sama, sehingga pemilihan algoritma harus diselaraskan dengan kebutuhan spesifik masalah (Bandyopadhyay, 2021). Oleh sebab itu, implementasi MCDM dalam studi ini berfungsi sebagai kerangka evaluasi formal yang menjamin bahwa penentuan rekomendasi lagu dilakukan secara konsisten, transparan, dan berdasarkan kriteria yang terukur.
Dalam penerapannya, tiap lagu diperlakukan sebagai kandidat alternatif, sementara atribut audio yang telah diekstraksi berperan sebagai kriteria penilaian. Preferensi situasional pengguna kemudian direpresentasikan melalui bobot yang memengaruhi kontribusi setiap kriteria terhadap hasil akhir. Melalui pendekatan ini, sistem mampu menghasilkan daftar rekomendasi yang selaras dengan konteks pengguna tanpa harus melakukan generalisasi yang kaku terhadap selera musik individu.
Di antara berbagai metode dalam kerangka MCDM, penelitian ini menggunakan metode Evaluation Based on Distance from Average Solution (EDAS) sebagai mekanisme pemeringkatan alternatif lagu. Metode ini dipilih karena mampu mengevaluasi alternatif berdasarkan deviasi terhadap nilai rata-rata setiap kriteria, sehingga proses penilaian dapat dilakukan secara sederhana namun tetap mempertimbangkan kontribusi masing-masing kriteria secara proporsional.
2.5 Metode Evaluation Based on Distance from Average Solution (EDAS)
Evaluation Based on Distance from Average Solution (EDAS) merupakan salah satu algoritma dalam kerangka Multi-Criteria Decision Making (MCDM) yang berfungsi menentukan alternatif optimal berdasarkan deviasi atau jaraknya terhadap solusi rata-rata (average solution). Metode ini hadir sebagai solusi untuk persoalan multikriteria yang memiliki atribut saling bertentangan, namun tetap menawarkan prosedur yang sederhana dan efisien dari sisi komputasi (Karim et al., 2022). Fleksibilitas serta stabilitas yang dimilikinya membuat EDAS banyak diimplementasikan dalam berbagai domain pengambilan keputusan (Sasmita et al., 2021).
Prinsip utama dari algoritma ini terletak pada penilaian setiap alternatif melalui pengukuran jarak relatif terhadap nilai rata-rata tiap kriteria. Jika diasumsikan terdapat matriks keputusan dengan m alternatif dan n kriteria, maka nilai rata-rata AV_j untuk setiap kriteria dapat dihitung menggunakan persamaan:
?AV?_j=1/m ?_(i=1)^m x_ij
Tahap berikutnya setelah perolehan nilai rata-rata adalah menghitung Jarak Positif dari Rata-Rata (Positive Distance from Average/PDA) dan Jarak Negatif dari Rata-Rata (Negative Distance from Average/NDA). Untuk kriteria dengan karakteristik keuntungan (benefit), formula yang digunakan adalah:
?PDA?_ij=((0,(X_ij-AV_j ))  )/(AV_j ),  ?NDA?_ij  =((0,(AV_j-X_ij ))  )/(AV_j )
Sedangkan untuk kriteria yang bersifat biaya (cost), perhitungan dilakukan dengan membalik selisih terhadap solusi rata-rata:
?PDA?_ij=  ((0,(AV_j-X_ij ))  )/(AV_j ),  ?NDA?_ij  =  ((0,(X_ij-AV_j ))  )/(AV_j )
Nilai PDA dan NDA kemudian diagregasikan dengan mempertimbangkan bobot masing-masing kriteria (w_j ) untuk memperoleh total jarak positif (?SP?_i ) dan total jarak negatif (?SN?_i ):
?SP?_i  = ?_(j=1)^n PDA_ij*w_j,        ?SN?_i= ?_(j=1)^n NDA_ij*w_j   
Untuk menghindari skala nilai yang berbeda antar alternatif, nilai (?SP?_i ) dan (?SN?_i ) kemudian dinormalisasi. Normalisasi terhadap jarak positif dilakukan dengan membagi setiap (?SP?_i ) terhadap nilai maksimum (?SP?_i ) yang diperoleh. Sementara itu, normalisasi jarak negatif dilakukan dengan mengurangkan rasio (?SN?_i ) terhadap nilai maksimum (?SN?_i ) dari satu, sehingga alternatif dengan jarak negatif lebih kecil memperoleh nilai yang lebih tinggi:
?NSP?_i ?=?(?SP?_i ?)/(max_i (SP_i ) ),  NSN_i=1-?SN?_i/(max_i (SN_i ) ) ??
Kemudian dikombinasikan untuk menghasilkan skor akhir atau Appraisal Score (?AS?_i ):
AS_i=1/2( NSP_i+NSN_i)
Hasil dari Appraisal Score inilah yang menjadi basis utama dalam memeringkat alternatif. Semakin besar nilai AS_i maka semakin tinggi pula tingkat kecocokan alternatif tersebut terhadap kriteria yang ditentukan. Dalam penelitian ini, lagu diposisikan sebagai alternatif, sementara fitur audio (seperti tempo, energy, valence, dan danceability) menjadi kriteria evaluasi. Preferensi situasional pengguna dikonversi menjadi bobot kriteria, sehingga hasil pemeringkatan mampu merefleksikan kebutuhan konteks secara personal.
Penggunaan EDAS dalam penelitian ini didasarkan pada keunggulannya dalam efisiensi komputasi serta kemampuannya menangani konflik antar-kriteria tanpa prosedur normalisasi yang rumit (Sasmita et al., 2021). Karakteristik tersebut relevan dengan permasalahan pada sistem rekomendasi musik, di mana setiap lagu memiliki berbagai parameter audio yang perlu dievaluasi secara simultan. Selain itu, stabilitas dan fleksibilitas metode EDAS dalam berbagai skenario pengambilan keputusan menjadikannya metode yang sesuai untuk melakukan pemeringkatan alternatif lagu berdasarkan preferensi pengguna (Biswas & Pamucar, 2023).
2.6 Natural Language Generation (NLG)
Natural Language Generation (NLG) merupakan sub-bidang dari Natural Language Processing (NLP) yang memiliki fokus utama pada konversi otomatis data terstruktur atau representasi sistem menjadi teks bahasa alami. Dalam skema data-to-text, NLG berperan penting dalam mentransformasikan data numerik maupun simbolik menjadi narasi yang koheren agar lebih mudah dipahami oleh pengguna. Pada awalnya, pendekatan NLG cenderung menggunakan metode berbasis aturan (rule-based) dan template yang sangat bergantung pada struktur kalimat yang kaku. Namun, pendekatan konvensional ini dinilai memiliki keterbatasan dalam hal fleksibilitas, variasi ekspresi, serta kemampuan untuk melakukan generalisasi terhadap data yang bersifat baru (Ampomah et al., 2022).
Seiring perkembangannya, implementasi NLG mulai memanfaatkan model deep learning serta pre-trained language models seperti T5 dan BART untuk memproduksi teks secara end-to-end tanpa ketergantungan pada template statis. Dalam metode ini, data terstruktur seperti tabel metrik diubah menjadi representasi linear (linearised input) yang kemudian diproses oleh model sequence-to-sequence menjadi teks deskriptif. Penggunaan model neural terbukti mampu menghasilkan penjelasan tekstual yang memiliki kefasihan linguistik tinggi sekaligus menjaga konsistensi informasi numerik yang bersumber dari tabel data (Ampomah et al., 2022). Namun pada banyak sistem praktis, termasuk sistem rekomendasi sederhana, pendekatan berbasis aturan atau template masih banyak digunakan karena lebih ringan dan mudah diimplementasikan. Hal ini menunjukkan bahwa NLG menjadi jembatan krusial antara data kuantitatif yang kompleks dengan pemahaman pengguna, khususnya bagi mereka yang tidak memiliki latar belakang teknis.
Dalam domain sistem rekomendasi, kebutuhan akan aspek transparansi dan interpretabilitas menjadi semakin mendesak seiring dengan meningkatnya kompleksitas model. Sistem rekomendasi masa kini tidak hanya bertugas menyajikan daftar item, tetapi juga harus mampu memberikan penjelasan yang personal serta informatif. Banyak sistem sebelumnya yang masih menggunakan template statis untuk menghasilkan penjelasan, sehingga ekspresivitas bahasa yang dihasilkan menjadi sangat terbatas. Untuk mengatasi kendala tersebut, penggunaan pendekatan hierarchical sequence-to-sequence diusulkan agar sistem dapat menghasilkan free-text natural language explanations yang bersifat personal dan kontekstual. Strategi ini memungkinkan sistem untuk lebih fleksibel dalam menyesuaikan narasi dengan karakteristik preferensi setiap pengguna (Chen et al., 2019).
Lebih lanjut, pemberian penjelasan (explanation generation) bukan sekadar elemen tambahan dalam sebuah rekomendasi, melainkan komponen kunci yang berkontribusi pada peningkatan kepercayaan, transparansi, serta kepuasan pengguna terhadap sistem. Melalui narasi yang menjelaskan alasan di balik sebuah rekomendasi, pengguna dapat memiliki pemahaman yang lebih baik mengenai proses pengambilan keputusan yang dilakukan oleh model (Chen et al., 2019).
Berdasarkan landasan tersebut, dalam penelitian ini NLG diimplementasikan sebagai lapisan interpretatif yang mengonversi hasil kalkulasi dari metode EDAS-yang berupa skor dan peringkat-menjadi narasi tekstual yang informatif. Dengan demikian, penerapan NLG tidak akan mengintervensi mekanisme perankingan dalam metode MCDM, melainkan berfungsi untuk mempresentasikan hasil rekomendasi dalam bentuk penjelasan tekstual yang membantu pengguna memahami alasan di balik pemilihan lagu dalam playlist yang dihasilkan.

Namun, kedua penelitian tersebut masih berfokus pada analisis empiris dan karakteristik populasi, serta belum mengimplementasikan mekanisme pemodelan preferensi individual dalam bentuk sistem rekomendasi yang terstruktur.
Selanjutnya, kajian sistematis oleh (Assun  o et al., 2022) mengidentifikasi beragam pendekatan rekomendasi berbasis emosi dan konteks, yang sebagian besar memanfaatkan machine learning dan histori interaksi. Meskipun demikian, pendekatan berbasis pembobotan eksplisit dari pengguna menggunakan metode multikriteria masih belum banyak dieksplorasi dalam domain rekomendasi musik berbasis aktivitas.
Dari sisi metode pengambilan keputusan multikriteria, penelitian oleh (Karim et al., 2022) serta (Biswas & Pamucar, 2023) menunjukkan bahwa metode EDAS mampu menghasilkan pemeringkatan alternatif secara sistematis dan objektif. Akan tetapi, penerapannya masih terbatas pada domain evaluasi layanan atau pemilihan alternatif statis, dan belum diarahkan pada sistem rekomendasi playlist musik berbasis preferensi kontekstual pengguna.
Sementara itu, penelitian oleh (Chen et al., 2019) dan (Ampomah et al., 2022) menegaskan potensi Natural Language Generation dalam menghasilkan penjelasan tekstual berbasis data terstruktur. Namun, integrasi NLG sebagai mekanisme penjelasan terhadap hasil pemeringkatan multikriteria dalam sistem rekomendasi musik berbasis konteks aktivitas masih terbatas.
Berdasarkan pemetaan tersebut, penelitian ini memposisikan diri pada integrasi pendekatan context-aware berbasis preferensi eksplisit pengguna, metode EDAS sebagai mekanisme pemeringkatan multikriteria, serta Natural Language Generation untuk menyajikan penjelasan rekomendasi dalam satu sistem rekomendasi playlist musik berbasis konteks aktivitas pengguna.

3.1 Rancangan Penelitian
3.1.1 Jenis dan Pendekatan Penelitian
Penelitian ini merupakan penelitian pengembangan sistem (research and development) yang bertujuan untuk merancang dan membangun sistem rekomendasi playlist musik berbasis konteks aktivitas pengguna. Sistem yang dikembangkan memanfaatkan metode Evaluation Based on Distance from Average Solution (EDAS) sebagai mekanisme pemeringkatan multikriteria serta Natural Language Generation (NLG) sebagai mekanisme penyajian hasil rekomendasi dalam bentuk narasi yang mudah dipahami.
Pendekatan penelitian difokuskan pada proses perancangan, implementasi, dan evaluasi sistem berbasis web yang dapat digunakan untuk menghasilkan rekomendasi playlist sesuai konteks aktivitas pengguna. Dalam penelitian ini, aktivitas pengguna, preferensi parameter audio, serta durasi playlist menjadi input utama yang diproses oleh sistem untuk menghasilkan daftar lagu yang telah diperingkat berdasarkan metode EDAS.
Sebagai penelitian pengembangan, hasil akhir dari penelitian ini tidak hanya berupa analisis konseptual, tetapi juga berupa artefak sistem yang dapat diuji fungsionalitas dan usability-nya. Oleh karena itu, penelitian ini mengintegrasikan pendekatan rekayasa perangkat lunak dengan metode evaluasi sistem untuk memastikan bahwa sistem yang dikembangkan sesuai dengan tujuan penelitian.
3.1.2 Model Pengembangan Sistem (SDLC - Waterfall)
Dalam pengembangan sistem rekomendasi playlist musik ini, digunakan kerangka kerja Software Development Life Cycle (SDLC) sebagai pendekatan pengembangan perangkat lunak. SDLC merupakan model konseptual yang menggambarkan tahapan sistematis dalam proses pengembangan perangkat lunak, mulai dari analisis kebutuhan hingga pemeliharaan sistem (Sommerville, 2011). Penggunaan SDLC bertujuan untuk memastikan bahwa proses pengembangan dilakukan secara terstruktur dan terdokumentasi dengan baik.
Model SDLC yang digunakan dalam penelitian ini adalah model Waterfall. Model Waterfall merupakan pendekatan pengembangan perangkat lunak yang bersifat linear dan berurutan, di mana setiap tahap harus diselesaikan sebelum melanjutkan ke tahap berikutnya (Sommerville, 2011).

Gambar 3.1 Diagram Waterfall (Sumber: medium.com/dot-intern, 2019)
Tahapan dalam model Waterfall meliputi:
1. Requirement Analysis
2. System Design
3. Implementation
4. Testing
5. Maintenance
Dalam konteks penelitian ini, tahapan Requirement Analysis mencakup identifikasi kebutuhan sistem rekomendasi, penentuan parameter audio sebagai kriteria dalam metode EDAS, serta spesifikasi input dan output sistem. Tahap System Design meliputi perancangan arsitektur sistem, perancangan basis data, serta perancangan algoritma EDAS dan modul NLG. Tahap Implementation merupakan realisasi rancangan menjadi aplikasi berbasis web. Tahap Testing meliputi pengujian fungsional sistem dan validasi hasil perhitungan metode EDAS serta keluaran NLG. Tahap Maintenance mencakup perbaikan dan penyempurnaan sistem setelah proses implementasi dan pengujian.
Pemilihan model Waterfall didasarkan pada pertimbangan bahwa kebutuhan sistem telah dirumuskan secara jelas sejak tahap awal penelitian, serta metode utama yang digunakan, yaitu EDAS dan NLG, telah ditentukan berdasarkan studi literatur sehingga tidak memerlukan eksplorasi metode secara iteratif. Selain itu, lingkup penelitian bersifat akademik dengan batasan yang terkontrol dan spesifikasi sistem relatif stabil.
Meskipun model Waterfall memiliki keterbatasan berupa kurang fleksibel terhadap perubahan kebutuhan di tengah proses pengembangan (Sommerville, 2011), pendekatan ini dinilai sesuai dengan karakteristik penelitian yang memiliki kebutuhan sistem yang telah terdefinisi dengan baik.
3.2 Tahapan Penelitian
Tahapan penelitian menggambarkan alur sistematis yang dilakukan dalam proses pengembangan sistem rekomendasi playlist musik berbasis konteks aktivitas pengguna. Tahapan ini disusun untuk memastikan bahwa proses penelitian berjalan secara terstruktur, mulai dari identifikasi permasalahan hingga evaluasi sistem yang dikembangkan.

Berdasarkan diagram tahapan penelitian, proses penelitian ini terdiri dari enam tahap utama, yaitu sebagai berikut:
1. Identifikasi Masalah
Tahap awal penelitian dilakukan dengan mengidentifikasi permasalahan pada sistem rekomendasi musik secara umum, khususnya terkait banyaknya pilihan lagu serta belum optimalnya pemanfaatan konteks aktivitas pengguna dalam pembentukan playlist.
Pada tahap ini dilakukan studi literatur dan pengamatan terhadap fenomena penggunaan musik dalam berbagai aktivitas untuk menemukan kesenjangan antara kebutuhan pengguna dan mekanisme rekomendasi yang tersedia. Hasil dari tahap ini adalah rumusan masalah yang menjadi dasar perancangan sistem.
2. Analisis Kebutuhan
Tahap analisis kebutuhan bertujuan untuk menentukan spesifikasi sistem yang akan dikembangkan. Pada tahap ini diidentifikasi kebutuhan fungsional dan non-fungsional sistem, termasuk kemampuan sistem dalam menerima input konteks aktivitas, menerima preferensi pengguna terhadap parameter audio, melakukan pemeringkatan lagu menggunakan metode EDAS, membentuk playlist sesuai durasi, serta menghasilkan penjelasan rekomendasi berbasis NLG.
Selain itu, ditentukan pula parameter audio yang digunakan sebagai kriteria dalam metode EDAS serta kebutuhan data yang diperlukan untuk mendukung proses perhitungan. Hasil dari tahap ini berupa dokumen spesifikasi kebutuhan sistem.
3. Perancangan Sistem
Tahap perancangan sistem dilakukan untuk menerjemahkan kebutuhan yang telah ditentukan ke dalam bentuk rancangan teknis. Pada tahap ini dilakukan perancangan arsitektur sistem, perancangan alur proses sistem, perancangan basis data, serta perancangan algoritma EDAS dan modul NLG. Perancangan ini menghasilkan blueprint sistem yang menjadi dasar implementasi.
4. Implementasi Sistem
Tahap implementasi merupakan proses realisasi rancangan sistem ke dalam bentuk aplikasi berbasis web. Pada tahap ini dilakukan pembangunan antarmuka pengguna, implementasi perhitungan metode EDAS untuk pemeringkatan lagu, pembentukan playlist berdasarkan durasi yang diinput pengguna, serta integrasi modul NLG untuk menghasilkan penjelasan rekomendasi. Tahap ini menghasilkan sistem yang dapat dijalankan dan diuji.
5. Pengujian Sistem
Tahap pengujian dilakukan untuk memastikan sistem berjalan sesuai dengan spesifikasi yang telah ditentukan. Pengujian dilakukan menggunakan metode black-box testing untuk memeriksa fungsi sistem, serta validasi terhadap hasil perhitungan metode EDAS guna memastikan akurasi pemeringkatan.
Selain itu, dilakukan pengujian terhadap keluaran modul NLG untuk memastikan kesesuaian antara hasil perhitungan dan narasi yang dihasilkan. Hasil dari tahap ini berupa laporan hasil pengujian sistem.
6. Evaluasi Sistem
Tahap evaluasi bertujuan untuk menilai efektivitas sistem yang telah dikembangkan. Pada tahap ini dilakukan analisis terhadap hasil pengujian, identifikasi keterbatasan sistem, serta penilaian terhadap pencapaian tujuan penelitian. Hasil akhir dari tahap ini menjadi dasar dalam penyusunan kesimpulan dan saran pengembangan lanjutan.
3.3 Objek Penelitian
Objek penelitian dalam penelitian ini adalah sistem rekomendasi playlist musik berbasis konteks aktivitas pengguna yang dikembangkan menggunakan metode Evaluation Based on Distance from Average Solution (EDAS) dan Natural Language Generation (NLG).
Secara operasional, objek penelitian mencakup dataset lagu yang digunakan sebagai alternatif dalam proses pemeringkatan, parameter audio sebagai kriteria evaluasi, serta data preferensi pengguna yang diperoleh melalui kuesioner.
3.3.1 Dataset Lagu
Dataset lagu yang digunakan dalam penelitian ini diperoleh dari sumber data daring yang menyediakan informasi daftar lagu populer beserta metadata terkait. Dataset ini dikumpulkan pada tanggal snapshot tertentu sebagai batasan waktu pengambilan data.
Setiap lagu dalam dataset memiliki atribut parameter audio yang digunakan sebagai kriteria dalam metode EDAS. Data tersebut kemudian diproses dan disiapkan dalam bentuk terstruktur agar dapat digunakan dalam pembentukan matriks keputusan. Dataset ini menjadi alternatif dalam proses pemeringkatan lagu untuk menghasilkan rekomendasi playlist sesuai konteks aktivitas pengguna.
3.3.2 Parameter Audio sebagai Kriteria
Parameter audio yang digunakan dalam penelitian ini berfungsi sebagai kriteria dalam metode EDAS. Parameter tersebut merepresentasikan karakteristik musikal dari setiap lagu.
Parameter yang digunakan meliputi:
? Tempo
? Energy
? Danceability
? Happiness
? Acousticness
? Instrumentalness
? Speechiness
? Popularity
Parameter-parameter tersebut digunakan untuk membentuk matriks keputusan dalam proses pemeringkatan. Nilai setiap parameter diperoleh dari metadata lagu dan selanjutnya diproses menggunakan metode EDAS untuk menghasilkan appraisal score. Pemilihan parameter audio didasarkan pada ketersediaan fitur kuantitatif standar dalam metadata lagu serta relevansinya terhadap karakteristik sebuah musik yang dapat dipersepsikan oleh pengguna dan digunakan dalam penelitian.
3.3.3 Responden dan Data Preferensi Pengguna
Selain dataset lagu, objek penelitian ini juga mencakup preferensi pengguna terhadap karakteristik musik dalam konteks aktivitas tertentu. Preferensi tersebut diperoleh melalui mekanisme kuesioner yang terintegrasi langsung di dalam sistem dan ditampilkan kepada pengguna pada saat pertama kali menggunakan aplikasi. Pendekatan ini digunakan untuk memperoleh preferensi pengguna secara eksplisit (explicit preference elicitation), sehingga sistem dapat menyesuaikan rekomendasi playlist dengan karakteristik musik yang diinginkan oleh pengguna sejak awal penggunaan sistem. 
Penggunaan mekanisme kuesioner dalam penelitian ini didasarkan pada pertimbangan bahwa sistem rekomendasi yang dikembangkan tidak memanfaatkan histori pemutaran musik pengguna sebagai sumber data preferensi. Oleh karena itu, preferensi pengguna perlu diperoleh secara langsung melalui interaksi awal dengan sistem. Pendekatan explicit preference elicitation memungkinkan sistem memperoleh gambaran awal mengenai kecenderungan preferensi musik pengguna sebelum proses rekomendasi dilakukan. Dengan demikian, proses rekomendasi tidak hanya bergantung pada karakteristik lagu yang tersedia dalam dataset, tetapi juga mempertimbangkan preferensi individual pengguna sebagai dasar dalam proses pemeringkatan.
Dalam penelitian ini, preferensi pengguna terhadap karakteristik musik dipandang sebagai variabel laten yang tidak dapat diukur secara langsung (Fernandes & Akhrani, 2022). Oleh karena itu, variabel tersebut dioperasionalisasikan ke dalam beberapa dimensi yang direpresentasikan oleh parameter audio lagu, yaitu BPM, energy, danceability, happiness, popularity, acousticness, instrumentalness, dan speechiness.
Setiap dimensi kemudian direpresentasikan melalui indikator yang menggambarkan persepsi pengguna terhadap karakteristik musik secara non-teknis. Indikator tersebut selanjutnya diukur melalui butir pernyataan dalam kuesioner berbasis skala Likert.

Dalam penelitian ini, pertanyaan kuesioner dirancang menggunakan bahasa yang mudah dipahami oleh pengguna umum dan tidak secara langsung menyebutkan istilah teknis parameter audio seperti tempo, energy, danceability, atau speechiness. Pendekatan ini dilakukan agar pengguna tidak perlu memahami konsep teknis karakteristik musik ketika menjawab pertanyaan. Sebaliknya, pengguna hanya diminta memberikan respons terhadap pernyataan yang menggambarkan suasana atau karakteristik musik secara umum. Dengan pendekatan tersebut, jawaban yang diberikan pengguna diharapkan dapat merepresentasikan preferensi musik secara lebih natural dan tidak dipengaruhi oleh pemahaman teknis terhadap parameter audio yang digunakan dalam sistem.
Jawaban pada kuesioner diukur menggunakan skala Likert lima tingkat, yang merepresentasikan tingkat persetujuan pengguna terhadap setiap pernyataan yang diberikan. Skala Likert merupakan metode pengukuran yang banyak digunakan dalam penelitian untuk mengukur sikap, persepsi, maupun preferensi responden terhadap suatu objek tertentu. Metode ini dipilih karena bentuk pernyataannya sederhana dan mudah dipahami oleh responden, sehingga dapat meminimalkan kesalahan interpretasi dalam proses pengisian kuesioner. Selain itu, skala Likert memungkinkan jawaban responden direpresentasikan dalam bentuk nilai numerik yang dapat digunakan dalam analisis kuantitatif.
Pemilihan skala Likert juga dipertimbangkan dengan membandingkannya terhadap beberapa metode pengukuran preferensi lain, seperti metode pemeringkatan langsung (ranking), perbandingan berpasangan (pairwise comparison), maupun pemberian skor numerik bebas. Metode pemeringkatan dan perbandingan berpasangan umumnya memerlukan proses evaluasi yang lebih kompleks karena responden harus membandingkan setiap alternatif secara langsung. Hal tersebut dinilai kurang sesuai dalam konteks penelitian ini karena jumlah karakteristik musik yang dinilai relatif banyak dan responden merupakan pengguna umum yang tidak selalu memiliki latar belakang pengetahuan musik. Oleh karena itu, skala Likert dipilih karena lebih sederhana, mudah dipahami, serta tetap mampu merepresentasikan tingkat preferensi pengguna secara terstruktur.
Dalam penelitian ini digunakan lima kategori jawaban pada skala Likert, yaitu:
1. Sangat Tidak Setuju
2. Tidak Setuju
3. Netral
4. Setuju
5. Sangat Setuju
Penggunaan lima tingkat skala dipilih karena mampu merepresentasikan variasi tingkat preferensi pengguna secara cukup rinci namun tetap menjaga kesederhanaan dalam proses pengisian kuesioner. Skala dengan jumlah tingkat yang lebih banyak, seperti tujuh atau sepuluh tingkat, memang dapat memberikan resolusi penilaian yang lebih tinggi, namun juga berpotensi meningkatkan kebingungan responden dalam membedakan pilihan jawaban yang tersedia. Oleh karena itu, skala lima tingkat dipandang sebagai kompromi yang seimbang antara tingkat detail penilaian dan kemudahan penggunaan bagi responden.
Berdasarkan pendekatan tersebut, disusun instrumen kuesioner yang terdiri dari beberapa butir pernyataan yang merepresentasikan indikator preferensi pengguna terhadap karakteristik musik secara non-teknis. Setiap pernyataan dirancang menggunakan bahasa yang natural dan mudah dipahami, sehingga pengguna dapat memberikan respons secara intuitif tanpa perlu memahami istilah teknis dalam parameter audio. 
Butir pernyataan dalam kuesioner ini disusun berdasarkan parameter audio yang digunakan sebagai kriteria dalam metode EDAS, sehingga setiap pernyataan secara konseptual merepresentasikan indikator dari parameter tersebut. Proses pemetaan antara pernyataan dan parameter dijelaskan lebih lanjut pada Bab IV. Pendekatan ini sejalan dengan prinsip perancangan instrumen penelitian, di mana variabel laten diturunkan menjadi indikator dan kemudian dioperasionalkan dalam bentuk butir pernyataan yang dapat diukur melalui kuesioner (Fernandes & Akhrani, 2022).
Adapun butir pernyataan yang digunakan dalam kuesioner adalah sebagai berikut:
Berikan jawaban terhadap setiap pernyataan berikut sesuai dengan preferensi musik yang ingin Anda dengarkan pada aktivitas yang telah Anda 	pilih sebelumnya. Bayangkan Anda sedang melakukan aktivitas tersebut, kemudian tentukan sejauh mana Anda setuju dengan pernyataan yang diberikan.
1. Saya lebih suka musik yang terasa cepat dan bikin suasana jadi lebih hidup. 
2. Saya lebih nyaman dengan musik yang santai dan tidak terburu-buru. 
3. Saya suka musik yang bisa bikin saya lebih semangat. 
4. Saya cenderung memilih musik yang tenang dan tidak terlalu kuat. 
5. Saya menikmati musik yang terasa aktif dan penuh energi. 
6. Saya suka musik yang bikin saya pengen ikut bergerak. 
7. Saya menikmati musik yang ritmenya enak untuk diikuti. 
8. Saya lebih menikmati musik yang terasa ceria dan positif. 
9. Saya biasanya lebih suka lagu yang sudah familiar di telinga saya. 
10. Saya suka musik yang terdengar alami seperti dimainkan dengan alat musik. 
11. Saya lebih suka musik tanpa banyak vokal. 
12. Saya menikmati musik yang fokus pada instrumen. 
13. Saya suka musik dengan lirik yang jelas dan menonjol. 
14. Saya lebih nyaman dengan musik yang tidak terlalu banyak kata-kata.
Setiap butir pernyataan dalam kuesioner tidak secara langsung merepresentasikan parameter audio, melainkan merepresentasikan indikator yang telah diturunkan dari masing-masing parameter. Dengan demikian, hubungan antara pertanyaan dan parameter bersifat tidak langsung, tetapi melalui indikator konseptual yang mencerminkan persepsi pengguna terhadap karakteristik musik. Dalam beberapa kasus, satu parameter direpresentasikan oleh lebih dari satu pernyataan untuk memperoleh pengukuran preferensi yang lebih stabil. Sebaliknya, parameter dengan karakteristik yang lebih spesifik direpresentasikan melalui satu pernyataan yang dianggap telah cukup mewakili indikator yang diukur. Pendekatan ini digunakan untuk menjaga keseimbangan antara akurasi pengukuran dan kesederhanaan instrumen kuesioner.
Nilai jawaban yang diberikan oleh pengguna kemudian dikonversi menjadi nilai numerik yang merepresentasikan tingkat preferensi terhadap karakteristik musik tertentu. Setiap nilai pada skala Likert diperlakukan sebagai representasi tingkat kecenderungan preferensi pengguna terhadap pernyataan yang diberikan. Nilai tersebut selanjutnya dipetakan ke dalam bobot kriteria yang digunakan dalam proses perhitungan metode Evaluation Based on Distance from Average Solution (EDAS). Dalam konteks metode pengambilan keputusan multikriteria, bobot kriteria merepresentasikan tingkat kepentingan relatif dari setiap parameter yang digunakan dalam proses evaluasi alternatif.
Melalui mekanisme ini, bobot kriteria dalam metode EDAS tidak ditentukan secara statis oleh sistem, melainkan dihasilkan secara dinamis berdasarkan preferensi pengguna yang diperoleh dari hasil kuesioner. Pendekatan tersebut memungkinkan sistem rekomendasi menghasilkan playlist yang lebih personal dan sesuai dengan karakteristik musik yang diinginkan oleh setiap pengguna dalam konteks aktivitas tertentu. Dengan demikian, rekomendasi yang dihasilkan tidak hanya bergantung pada karakteristik lagu dalam dataset, tetapi juga mempertimbangkan preferensi individual pengguna sebagai bagian dari proses pengambilan keputusan.

BAB IV
HASIL DAN PEMBAHASAN
4.1 Analisis Kebutuhan Sistem
Analisis kebutuhan sistem pada penelitian ini diturunkan dari rumusan masalah dan tujuan penelitian (Bab I), kajian literatur terkait sistem rekomendasi musik berbasis konteks, metode EDAS, dan NLG (Bab II), serta kebutuhan operasional metode EDAS dalam melakukan pemeringkatan multikriteria. Hasil analisis kebutuhan ini menjadi dasar dalam proses perancangan dan implementasi sistem.
Kebutuhan sistem yang telah diidentifikasi terdiri atas kebutuhan fungsional, kebutuhan non-fungsional, serta spesifikasi input dan output sistem.
4.1.1 Kebutuhan Fungsional
Kebutuhan fungsional menggambarkan fungsi utama yang harus dimiliki oleh sistem agar dapat menghasilkan rekomendasi playlist sesuai dengan tujuan penelitian. Berdasarkan hasil analisis, sistem yang dikembangkan memiliki kebutuhan fungsional sebagai berikut:
1. Sistem mampu menerima input konteks aktivitas pengguna.
2. Sistem mampu menerima input preferensi pengguna melalui mekanisme kuesioner yang terintegrasi dalam sistem.
3. Sistem mampu menerima input durasi playlist yang diinginkan pengguna.
4. Sistem mampu mengakses dan memproses dataset lagu beserta parameter audio sebagai kriteria dalam metode EDAS.
5. Sistem mampu membentuk matriks keputusan berdasarkan alternatif lagu dan kriteria audio.
6. Sistem mampu menghitung nilai average solution pada setiap kriteria.
7. Sistem mampu menghitung Positive Distance from Average (PDA) dan Negative Distance from Average (NDA).
8. Sistem mampu melakukan normalisasi dan menghasilkan appraisal score untuk setiap alternatif lagu.
9. Sistem mampu menghasilkan peringkat lagu berdasarkan nilai appraisal score.
10. Sistem mampu menyusun playlist berdasarkan hasil pemeringkatan serta menyesuaikan total durasi dengan input pengguna.
11. Sistem mampu menghasilkan penjelasan rekomendasi dalam bentuk narasi menggunakan modul Natural Language Generation (NLG).
12. Sistem mampu menampilkan hasil playlist dan penjelasan rekomendasi melalui antarmuka berbasis web.
4.1.2 Kebutuhan Non-Fungsional
Kebutuhan non-fungsional menjelaskan karakteristik kualitas sistem yang mendukung kinerja dan kenyamanan penggunaan. Berdasarkan analisis yang dilakukan, kebutuhan non-fungsional sistem adalah sebagai berikut:
1. Sistem dikembangkan berbasis web sehingga dapat diakses melalui peramban tanpa instalasi tambahan.
2. Sistem memiliki antarmuka yang sederhana dan mudah dipahami oleh pengguna.
3. Proses perhitungan rekomendasi dilakukan secara efisien sehingga waktu respons sistem tetap dalam batas yang wajar.
4. Sistem mampu melakukan perhitungan metode EDAS secara konsisten dan akurat.
5. Data preferensi pengguna tidak disimpan secara permanen kecuali diperlukan untuk keperluan penelitian.
4.1.3 Spesifikasi Input dan Output Sistem
Untuk memperjelas batasan operasional sistem, berikut adalah spesifikasi input dan output sistem yang dikembangkan.
1. Input Sistem:
? Jenis aktivitas yang dipilih pengguna.
? Preferensi pengguna terhadap karakteristik musik (diperoleh melalui kuesioner terintegrasi).
? Durasi playlist yang diinginkan.
? Dataset lagu beserta parameter audio sebagai kriteria.
2. Output Sistem:
? Daftar lagu yang telah diperingkat berdasarkan metode EDAS.
? Playlist dengan total durasi mendekati input pengguna.
? Penjelasan rekomendasi dalam bentuk narasi yang dihasilkan oleh modul NLG.
4.2 Perancangan Sistem
4.2.1 Perancangan Proses Sistem (BPMN)
Perancangan proses sistem menggambarkan alur interaksi antara pengguna dan sistem dalam menghasilkan rekomendasi playlist berbasis konteks aktivitas. Proses dimulai ketika pengguna mengakses aplikasi melalui antarmuka web. Sistem kemudian menampilkan halaman input yang berisi pilihan konteks aktivitas serta durasi playlist yang diinginkan.
Setelah pengguna memilih konteks aktivitas dan durasi, sistem menampilkan formulir preferensi yang berisi sejumlah pertanyaan terkait karakteristik musik. Pertanyaan tersebut dirancang dalam bentuk yang mudah dipahami pengguna, misalnya preferensi terhadap musik yang lebih energik, lebih tenang, lebih ceria, atau minim vokal. Jawaban pengguna kemudian diproses oleh sistem dan dikonversi menjadi bobot numerik untuk setiap parameter audio yang digunakan sebagai kriteria evaluasi.
Setelah seluruh input diterima, sistem memproses dataset lagu dengan memanfaatkan parameter audio sebagai kriteria penilaian. Proses evaluasi dilakukan menggunakan metode Evaluation Based on Distance from Average Solution (EDAS) untuk menghasilkan nilai penilaian bagi setiap lagu. Hasil perhitungan tersebut kemudian digunakan untuk menyusun peringkat lagu dari nilai tertinggi ke terendah.
Berdasarkan hasil pemeringkatan tersebut, sistem melakukan proses seleksi lagu secara berurutan dengan mempertimbangkan nilai ranking serta durasi masing-masing lagu. Lagu-lagu dengan peringkat tertinggi dipilih hingga total durasi playlist mendekati durasi aktivitas yang dimasukkan oleh pengguna.
Setelah playlist terbentuk, sistem menyimpan informasi playlist serta daftar lagu yang terpilih ke dalam basis data. Informasi ini mencakup identitas playlist, sesi yang menghasilkan playlist tersebut, serta urutan lagu dan nilai appraisal score hasil perhitungan EDAS.
Selanjutnya sistem mengaktifkan modul Natural Language Generation (NLG) untuk menghasilkan penjelasan naratif berdasarkan hasil pemeringkatan dan karakteristik lagu yang direkomendasikan. Sistem kemudian menampilkan playlist rekomendasi beserta penjelasan kepada pengguna melalui antarmuka aplikasi.

4.2.5 Perancangan Algoritma EDAS
Metode Evaluation based on Distance from Average Solution (EDAS) digunakan dalam penelitian ini untuk melakukan proses pemeringkatan lagu berdasarkan kesesuaian karakteristik audio terhadap preferensi pengguna. Metode ini dipilih karena mampu mengevaluasi alternatif berdasarkan beberapa kriteria secara simultan serta menghasilkan nilai preferensi yang dapat digunakan untuk menentukan alternatif terbaik.
Dalam sistem yang dikembangkan pada penelitian ini, setiap lagu diperlakukan sebagai alternatif, sedangkan karakteristik audio lagu digunakan sebagai kriteria penilaian. Preferensi pengguna terhadap karakteristik musik diperoleh melalui kuesioner yang diisi oleh pengguna setelah memasukkan konteks aktivitas yang akan dilakukan. Jawaban dari kuesioner tersebut kemudian dikonversi menjadi bobot kriteria yang digunakan dalam proses perhitungan metode EDAS. Melalui pendekatan ini, sistem dapat menghasilkan rekomendasi playlist musik yang disesuaikan secara dinamis berdasarkan preferensi dan konteks aktivitas pengguna. Langkah-langkah metode EDAS dalam penelitian ini dilakukan secara berurutan sebagai berikut.
a. Penentuan Kriteria Penilaian Lagu
Kriteria yang digunakan dalam proses pemeringkatan lagu pada penelitian ini berasal dari parameter audio yang menggambarkan karakteristik musikal dari setiap lagu. Parameter tersebut diperoleh dari dataset fitur audio lagu yang diperoleh dari platform analisis musik daring, sehingga setiap lagu memiliki nilai kuantitatif yang merepresentasikan karakteristik audionya.
Dalam penelitian ini digunakan delapan parameter audio utama sebagai kriteria penilaian, yaitu BPM, energy, danceability, happiness, popularity, acousticness, instrumentalness, dan speechiness. Parameter-parameter tersebut dipilih karena mampu merepresentasikan berbagai aspek karakteristik musik yang berpengaruh terhadap pengalaman mendengarkan pengguna. Sebagai contoh, parameter energy merepresentasikan tingkat intensitas musik, danceability menggambarkan kesesuaian ritme lagu untuk aktivitas bergerak, happiness merepresentasikan nuansa emosional lagu, sedangkan instrumentalness dan speechiness menggambarkan dominasi instrumen maupun unsur vokal dalam komposisi lagu.
Pemilihan parameter tersebut juga mempertimbangkan kesesuaiannya dengan metode pengambilan keputusan multikriteria seperti EDAS. Metode EDAS memerlukan nilai kriteria yang bersifat numerik agar dapat dihitung jaraknya terhadap solusi rata-rata setiap kriteria. Oleh karena itu, parameter audio yang digunakan pada penelitian ini dipilih dari fitur yang memiliki nilai kuantitatif dan dapat dibandingkan secara langsung antar lagu.

b. Penentuan Bobot Kriteria dan Jenis Kriteria Berdasarkan Preferensi Pengguna
Pendekatan yang digunakan dalam penelitian ini bersifat user-centric, sehingga tingkat kepentingan masing-masing parameter tidak ditentukan secara tetap oleh pakar ataupun berdasarkan asumsi peneliti, melainkan ditentukan berdasarkan preferensi pengguna yang diperoleh melalui kuesioner preferensi musik. Pendekatan ini digunakan agar sistem dapat menghasilkan rekomendasi playlist yang lebih personal dan sesuai dengan karakteristik musik yang diinginkan pengguna pada konteks aktivitas tertentu.
Preferensi pengguna dikumpulkan melalui kuesioner yang disusun dalam bentuk pernyataan sederhana mengenai karakteristik musik, seperti kecenderungan terhadap musik yang energik, musik dengan tempo cepat, musik yang ceria, musik yang minim vokal, atau musik dengan unsur akustik yang dominan. Pertanyaan dirancang menggunakan bahasa yang mudah dipahami oleh pengguna umum dan tidak secara langsung menyebutkan istilah teknis parameter audio. Dengan pendekatan tersebut, pengguna dapat memberikan jawaban secara lebih natural tanpa harus memahami konsep teknis seperti BPM, energy, atau speechiness.
Jawaban pengguna diukur menggunakan skala Likert lima tingkat, yaitu Sangat Tidak Setuju, Tidak Setuju, Netral, Setuju, dan Sangat Setuju, yang masing-masing dikonversi ke dalam nilai numerik 1 sampai 5. Nilai yang lebih tinggi menunjukkan tingkat preferensi yang lebih kuat terhadap karakteristik musik tertentu, sedangkan nilai yang lebih rendah menunjukkan tingkat preferensi yang lebih rendah. Nilai netral diperlakukan sebagai nilai moderat yang menunjukkan bahwa pengguna tidak memiliki kecenderungan preferensi yang dominan terhadap karakteristik musik yang dimaksud.
Setiap butir pernyataan dalam kuesioner tidak secara langsung merepresentasikan parameter audio, melainkan merepresentasikan indikator yang telah diturunkan dari masing-masing parameter. Dengan demikian, hubungan antara pertanyaan dan parameter bersifat tidak langsung, tetapi melalui indikator konseptual yang mencerminkan persepsi pengguna terhadap karakteristik musik. Pendekatan ini digunakan agar pengguna dapat memberikan respons secara lebih natural tanpa perlu memahami istilah teknis dalam parameter audio.
Setiap pertanyaan pada kuesioner kemudian dipetakan ke parameter audio tertentu yang digunakan sebagai kriteria dalam metode EDAS.

Apabila suatu parameter diwakili oleh lebih dari satu pertanyaan, maka skor preferensi parameter tersebut diperoleh dengan menghitung rata-rata nilai jawaban dari seluruh pertanyaan yang merepresentasikan parameter tersebut. Secara matematis, nilai preferensi untuk setiap parameter dihitung sebagai rata-rata dari seluruh nilai pertanyaan yang merepresentasikan parameter tersebut. Dengan demikian, setiap parameter audio akan memiliki satu skor preferensi yang merepresentasikan tingkat kepentingannya bagi pengguna.
Jumlah pertanyaan yang merepresentasikan setiap parameter tidak ditentukan secara seragam, melainkan disesuaikan dengan kompleksitas karakteristik yang diukur. Parameter dengan representasi konsep yang lebih luas seperti energy dan BPM direpresentasikan melalui lebih dari satu pertanyaan untuk memperoleh pengukuran preferensi yang lebih stabil. Sebaliknya, parameter dengan karakteristik yang lebih spesifik seperti popularity atau acousticness direpresentasikan melalui satu pertanyaan yang dianggap telah cukup mewakili indikator yang diukur. Pendekatan ini sejalan dengan prinsip penyusunan instrumen kuesioner, di mana jumlah indikator tidak harus sama untuk setiap dimensi, tetapi disesuaikan dengan kebutuhan representasi konsep.
Setelah skor preferensi seluruh parameter diperoleh, langkah berikutnya adalah melakukan normalisasi nilai skor tersebut untuk menghasilkan bobot kriteria. Normalisasi dilakukan dengan membagi skor setiap parameter dengan total seluruh skor parameter sehingga diperoleh bobot yang berada pada rentang 0 sampai 1 dan memiliki jumlah total sebesar 1. Bobot inilah yang selanjutnya digunakan dalam proses perhitungan metode EDAS.
Secara matematis, bobot setiap kriteria dihitung menggunakan persamaan sebagai berikut:
w_j  =S_j/(? S_j )
Keterangan:
w_j
: bobot kriteria ke-j
S_j
: skor preferensi parameter ke-j
? S_j 
: total skor seluruh parameter
Dengan demikian, proses konversi preferensi pengguna melalui kuesioner menghasilkan dua komponen utama dalam metode EDAS, yaitu bobot kriteria dan jenis kriteria (benefit atau cost). Kedua komponen ini selanjutnya digunakan dalam proses evaluasi alternatif lagu pada tahap perhitungan metode EDAS. 
Pemilihan pendekatan ini dilakukan karena lebih sesuai untuk sistem rekomendasi musik yang ditujukan kepada pengguna umum. Metode pembobotan seperti Analytical Hierarchy Process (AHP) umumnya menggunakan perbandingan berpasangan antar kriteria. Pendekatan tersebut kurang sesuai dalam konteks penelitian ini karena pengguna harus membandingkan parameter audio yang bersifat teknis satu per satu. Dengan delapan parameter audio, pengguna harus melakukan 28 perbandingan berpasangan, yang berpotensi meningkatkan beban kognitif pengguna serta menurunkan kemudahan penggunaan sistem. Oleh karena itu, penelitian ini menggunakan pendekatan agregasi skor kuesioner berbasis skala Likert yang lebih sederhana, mudah dipahami, dan tetap mampu menghasilkan bobot kriteria yang dibutuhkan dalam metode EDAS.
Selain digunakan untuk menentukan bobot kriteria, preferensi pengguna juga digunakan untuk menentukan jenis kriteria, yaitu benefit atau cost. Dalam metode EDAS, kriteria benefit merupakan kriteria yang nilainya diharapkan semakin tinggi untuk menghasilkan alternatif yang lebih baik, sedangkan kriteria cost merupakan kriteria yang nilainya diharapkan semakin rendah. Pada penelitian ini, klasifikasi benefit dan cost tidak ditentukan secara statis pada tingkat sistem, tetapi ditentukan berdasarkan kecenderungan preferensi pengguna terhadap karakteristik musik yang ditanyakan dalam kuesioner.

Untuk nilai netral, parameter tidak secara langsung diklasifikasikan sebagai benefit atau cost secara ekstrem, melainkan tetap diproses sebagai nilai moderat dalam perhitungan tanpa membalik arah preferensi utama pengguna.
Apabila jawaban pengguna menunjukkan kecenderungan menyukai tingkat karakteristik yang lebih tinggi pada suatu parameter, maka parameter tersebut diperlakukan sebagai kriteria benefit. Sebaliknya, apabila jawaban pengguna menunjukkan kecenderungan menghindari atau kurang menyukai dominasi karakteristik tertentu, maka parameter tersebut diklasifikasikan sebagai kriteria cost, yaitu kriteria yang nilainya diharapkan semakin rendah.
Apabila nilai rata-rata preferensi pengguna berada pada kondisi netral, maka parameter tidak mengubah arah kriteria dan tetap diproses menggunakan nilai preferensi sebagaimana adanya sebagai nilai moderat dalam perhitungan. Dengan demikian, nilai netral tetap berkontribusi dalam pembentukan bobot kriteria, namun tidak digunakan untuk membalik arah kecenderungan preferensi pengguna. Dengan mekanisme ini, sistem dapat menyesuaikan proses evaluasi lagu dengan preferensi pengguna secara lebih fleksibel dan kontekstual.
Sebagai contoh, dalam konteks aktivitas olahraga pengguna mungkin lebih menyukai musik dengan tingkat energy atau BPM yang tinggi, sehingga parameter tersebut diperlakukan sebagai kriteria benefit. Sebaliknya, pada aktivitas yang memerlukan konsentrasi seperti belajar, pengguna mungkin lebih menyukai musik dengan tingkat speechiness yang rendah, sehingga parameter tersebut dapat diperlakukan sebagai kriteria cost.

d. Perhitungan Average Solution
Langkah selanjutnya dalam metode EDAS adalah menghitung nilai rata-rata dari setiap kriteria yang disebut sebagai Average Solution. Nilai rata-rata ini digunakan sebagai acuan untuk mengevaluasi posisi setiap alternatif terhadap kondisi rata-rata seluruh alternatif pada masing-masing kriteria.
Perhitungan Average Solution dilakukan dengan menghitung nilai rata-rata untuk setiap kriteria berdasarkan seluruh alternatif lagu yang terdapat pada matriks keputusan. Dengan kata lain, nilai rata-rata setiap parameter audio diperoleh dari keseluruhan dataset lagu yang digunakan dalam sistem. Nilai tersebut merepresentasikan kondisi umum dari karakteristik musik pada dataset yang digunakan dalam penelitian ini.
Nilai rata-rata ini kemudian digunakan sebagai baseline evaluasi dalam metode EDAS. Melalui baseline tersebut, sistem dapat menentukan apakah nilai suatu alternatif berada di atas atau di bawah nilai rata-rata pada setiap kriteria.
Hasil perhitungan Average Solution selanjutnya digunakan dalam tahap perhitungan Positive Distance from Average (PDA) dan Negative Distance from Average (NDA) untuk setiap alternatif lagu, sebagaimana dijelaskan pada metode EDAS pada Bab sebelumnya.
Perhitungan nilai rata-rata setiap kriteria dalam metode EDAS dapat dinyatakan menggunakan persamaan berikut.
AV_j=1/m  ?_(i=1)^m  x_ij
Keterangan:

:
nilai rata-rata untuk kriteria ke-j
x_ij
:
nilai alternatif ke-i pada kriteria ke-j
m

e. Perhitungan Positive Distance from Average (PDA) dan Negative Distance from Average (NDA)
Setelah nilai rata-rata setiap kriteria diperoleh, langkah berikutnya adalah menghitung nilai Positive Distance from Average (PDA) dan Negative Distance from Average (NDA) untuk setiap alternatif lagu. Perhitungan ini bertujuan untuk mengukur seberapa jauh nilai suatu alternatif berada di atas atau di bawah nilai rata-rata pada masing-masing kriteria.
Perhitungan nilai Positive Distance from Average (PDA) dan Negative Distance from Average (NDA) dalam metode EDAS dinyatakan sebagai berikut.
Untuk kriteria benefit:
PDA_ij=((max(0, x_ij-AV_j )))/AV_j ,  NDA_ij=((max(0, AV_j-x_ij )))/AV_j 
Untuk kriteria cost:
PDA_ij=((max(0, AV_j-x_ij )))/AV_j ,  NDA_ij=((max(0, x_ij-AV_j )))/AV_j   
Keterangan:
PDA_ij
: jarak positif alternatif ke-i pada kriteria ke-j
NDA_ij
: jarak negatif alternatif ke-i pada kriteria ke-j
x_ij
: nilai alternatif ke-i pada kriteria ke-j
AV_j
: nilai rata-rata kriteria ke-j
Nilai PDA menunjukkan jarak positif suatu alternatif terhadap nilai rata-rata pada suatu kriteria. Untuk kriteria benefit, PDA akan bernilai positif ketika nilai alternatif berada di atas nilai rata-rata. Sementara itu, nilai NDA menunjukkan jarak negatif suatu alternatif terhadap nilai rata-rata. Untuk kriteria benefit, NDA akan bernilai positif ketika nilai alternatif berada di bawah nilai rata-rata. Pada kriteria cost, mekanisme perhitungan dilakukan dengan arah yang berlawanan, karena nilai yang lebih rendah justru lebih diharapkan.
Dengan demikian, perhitungan PDA dan NDA dilakukan dengan mempertimbangkan dua hal sekaligus, yaitu nilai parameter audio pada dataset lagu dan jenis kriteria (benefit atau cost) yang telah ditentukan berdasarkan preferensi pengguna.
f. Perhitungan Total Nilai Positif dan Negatif (SP dan SN)
Nilai PDA dan NDA yang diperoleh pada setiap kriteria selanjutnya diagregasikan dengan memperhatikan bobot kriteria. Hasil agregasi ini menghasilkan dua nilai total untuk setiap alternatif, yaitu Weighted Sum of Positive Distances (SP) dan Weighted Sum of Negative Distances (SN).
Nilai SP diperoleh dari penjumlahan berbobot seluruh nilai PDA pada setiap kriteria, sedangkan nilai SN diperoleh dari penjumlahan berbobot seluruh nilai NDA pada setiap kriteria. Melalui langkah ini, sistem dapat memperoleh total kontribusi positif dan total kontribusi negatif dari setiap alternatif lagu terhadap keseluruhan kriteria yang digunakan.
Total nilai jarak positif dan negatif pada setiap alternatif dihitung menggunakan penjumlahan berbobot terhadap seluruh kriteria sebagai berikut.
Rumus SP:
SP_i  =?_(j=1)^n  w_j  PDA_ij
Rumus SN:
SN_i  =?_(j=1)^n  w_j  NDA_ij
Keterangan:
SP_i
: total jarak positif alternatif ke-i
SN_i
: total jarak negatif alternatif ke-i
w_j
: bobot kriteria ke-j
n
: jumlah kriteria
g. Normalisasi Nilai SP dan SN
Setelah nilai SP dan SN diperoleh untuk setiap alternatif, langkah berikutnya adalah melakukan normalisasi terhadap kedua nilai tersebut. Normalisasi dilakukan agar nilai SP dan SN berada pada skala yang sebanding sehingga dapat digunakan untuk menghitung skor akhir setiap alternatif.
Nilai SP dinormalisasi untuk memperoleh nilai positif ternormalisasi, sedangkan nilai SN dinormalisasi untuk memperoleh nilai negatif ternormalisasi. Proses normalisasi ini penting karena hasil agregasi berbobot dapat memiliki rentang nilai yang berbeda antar alternatif, sehingga perlu disetarakan sebelum dihitung menjadi nilai akhir.
Normalisasi SP:
NSP_i=SP_i/max(SP_i ) 
Normalisasi SN:
NSN_i=1-(SN_i/max(SN_i ) )
Keterangan:
NSP_i
: nilai positif ternormalisasi
NSN_i
: nilai negatif ternormalisasi
h. Perhitungan Appraisal Score dan Pemeringkatan Lagu
Langkah selanjutnya adalah menghitung nilai akhir setiap alternatif yang disebut sebagai Appraisal Score. Nilai ini diperoleh dari kombinasi nilai positif ternormalisasi dan nilai negatif ternormalisasi yang dihasilkan pada tahap sebelumnya. 
Nilai akhir setiap alternatif dalam metode EDAS dihitung menggunakan nilai appraisal score yang merupakan kombinasi dari nilai positif ternormalisasi dan nilai negatif ternormalisasi.
Rumus:
?AS?_i=(?NSP?_i+?NSN?_i)/2
Keterangan:
AS_i
: appraisal score alternatif ke-i
NSP_i
: nilai positif ternormalisasi
NSN_i
: nilai negatif ternormalisasi
Nilai Appraisal Score merepresentasikan tingkat kesesuaian setiap alternatif lagu terhadap kriteria yang digunakan dalam metode EDAS. Semakin tinggi nilai Appraisal Score yang diperoleh suatu lagu, maka semakin tinggi pula tingkat kecocokan lagu tersebut terhadap preferensi pengguna berdasarkan parameter audio yang digunakan dalam sistem.
Nilai Appraisal Score kemudian digunakan sebagai dasar dalam proses pemeringkatan alternatif lagu. Sistem menyusun seluruh lagu dalam dataset berdasarkan nilai Appraisal Score dari yang tertinggi hingga terendah.
Setelah proses pemeringkatan selesai dilakukan, sistem membentuk playlist musik dengan memilih lagu secara berurutan berdasarkan nilai Appraisal Score tertinggi. Proses pemilihan lagu dilakukan hingga total durasi lagu yang terpilih mendekati atau sedikit melebihi durasi aktivitas yang telah diinput oleh pengguna sebelumnya. Dengan mekanisme ini, playlist yang dihasilkan tidak hanya mempertimbangkan tingkat kesesuaian karakteristik musik, tetapi juga menyesuaikan dengan durasi aktivitas pengguna. Playlist yang telah terbentuk selanjutnya dapat disimpan sebagai hasil rekomendasi pada sesi penggunaan sistem, termasuk informasi urutan lagu dan nilai evaluasi yang diperoleh dari proses pemeringkatan.

4.2.6 Perancangan Modul Natural Language Generation (NLG)
Modul Natural Language Generation (NLG) pada sistem ini dirancang untuk menghasilkan penjelasan naratif terhadap hasil rekomendasi playlist yang diperoleh dari proses perhitungan metode EDAS. Tujuan utama dari modul ini adalah membantu pengguna memahami alasan pemilihan lagu dalam playlist yang direkomendasikan oleh sistem. Dengan adanya penjelasan dalam bentuk bahasa alami, pengguna tidak hanya menerima daftar lagu yang direkomendasikan, tetapi juga memperoleh gambaran mengenai karakteristik musik yang menjadi dasar rekomendasi tersebut.
Dalam sistem yang dikembangkan pada penelitian ini, modul NLG tidak mempengaruhi proses perhitungan metode EDAS. Seluruh proses pemeringkatan lagu tetap ditentukan oleh algoritma EDAS berdasarkan parameter audio lagu serta bobot preferensi yang diperoleh dari pengguna. Modul NLG hanya berfungsi sebagai lapisan interpretasi yang mengubah hasil pemeringkatan tersebut menjadi penjelasan naratif yang lebih mudah dipahami oleh pengguna.
Proses pembentukan narasi dimulai ketika pengguna pertama kali menggunakan sistem dengan memasukkan konteks aktivitas yang akan dilakukan, seperti jenis aktivitas, waktu pelaksanaan aktivitas, suasana hati pengguna, serta durasi aktivitas yang diinginkan. Setelah konteks aktivitas ditentukan, sistem akan menampilkan kuesioner preferensi musik yang bertujuan untuk mengetahui kecenderungan karakteristik musik yang diinginkan oleh pengguna pada konteks aktivitas tersebut. Jawaban pengguna terhadap kuesioner tersebut kemudian dikonversi oleh sistem menjadi bobot preferensi yang akan digunakan dalam proses perhitungan metode EDAS.
Setelah bobot preferensi diperoleh, sistem membentuk matriks keputusan berdasarkan dataset lagu yang telah dikumpulkan sebelumnya. Dataset tersebut berisi nilai parameter audio dari setiap lagu, seperti BPM, energy, danceability, happiness, popularity, acousticness, instrumentalness, dan speechiness. Matriks keputusan tersebut kemudian digunakan dalam proses perhitungan metode EDAS yang meliputi perhitungan Average Solution, Positive Distance from Average (PDA), dan Negative Distance from Average (NDA). Hasil dari proses tersebut adalah nilai Appraisal Score untuk setiap lagu dalam dataset.
Nilai Appraisal Score kemudian digunakan untuk menyusun pemeringkatan lagu dari nilai tertinggi hingga terendah. Berdasarkan urutan tersebut, sistem membentuk playlist dengan memilih lagu secara berurutan hingga total durasi lagu yang terpilih mendekati atau sedikit melebihi durasi aktivitas yang telah dimasukkan oleh pengguna. Playlist yang dihasilkan pada tahap ini merupakan hasil akhir dari proses rekomendasi yang dilakukan oleh metode EDAS. Playlist tersebut selanjutnya dapat disimpan sebagai hasil rekomendasi pada sesi aktif dan digunakan sebagai dasar dalam penyusunan informasi untuk modul NLG.
Setelah playlist terbentuk, sistem menyusun struktur informasi ringkasan rekomendasi yang akan digunakan sebagai masukan bagi modul NLG. Informasi tersebut meliputi konteks aktivitas pengguna, karakteristik umum dari lagu-lagu yang terpilih dalam playlist, serta alasan pemilihan lagu berdasarkan parameter audio yang dominan. Struktur informasi tersebut kemudian digunakan sebagai dasar dalam penyusunan prompt yang dikirimkan ke model bahasa melalui layanan API untuk menghasilkan narasi penjelasan rekomendasi playlist dalam bentuk bahasa alami.
Narasi yang dihasilkan oleh modul NLG kemudian ditampilkan bersamaan dengan playlist yang direkomendasikan kepada pengguna. Melalui mekanisme ini, pengguna tidak hanya memperoleh daftar lagu yang sesuai dengan preferensinya, tetapi juga memperoleh penjelasan mengenai karakteristik playlist yang dihasilkan oleh sistem serta alasan mengapa lagu-lagu tersebut direkomendasikan pada konteks aktivitas yang dipilih.

Melalui wireframe ini, dapat digambarkan bagaimana pengguna memasukkan konteks aktivitas, mengisi kuesioner preferensi musik, menjalankan proses rekomendasi, hingga memperoleh hasil rekomendasi playlist yang dihasilkan oleh sistem.

Tampilan rancangan halaman input konteks aktivitas ditunjukkan pada gambar berikut.

2. Halaman Kuesioner Preferensi Musik
Setelah pengguna memasukkan konteks aktivitas pada halaman sebelumnya, sistem akan menampilkan halaman kuesioner preferensi musik. Halaman ini bertujuan untuk mengumpulkan informasi mengenai preferensi pengguna terhadap karakteristik musik yang diinginkan pada aktivitas yang akan dilakukan.
Pada bagian atas halaman ditampilkan judul "Kuesioner Preferensi Musik" yang diikuti dengan teks penjelasan singkat bahwa jawaban pengguna akan membantu sistem menentukan karakteristik musik yang sesuai. Informasi ini bertujuan memberikan pemahaman kepada pengguna mengenai tujuan pengisian kuesioner dalam proses rekomendasi playlist.
Di bagian utama halaman ditampilkan beberapa pertanyaan kuesioner yang menggunakan skala Likert. Setiap pertanyaan memiliki lima pilihan jawaban yang merepresentasikan tingkat persetujuan pengguna terhadap pernyataan yang diberikan. Skala jawaban tersebut terdiri dari nilai 1 hingga 5 yang mewakili rentang Sangat Tidak Setuju hingga Sangat Setuju.
Pertanyaan dalam kuesioner dirancang untuk menggambarkan preferensi pengguna terhadap karakteristik musik tertentu, seperti tempo musik, tingkat energi lagu, suasana emosional lagu, dominasi vokal, kemudahan mengikuti ritme musik, serta unsur akustik pada lagu. Melalui pertanyaan-pertanyaan tersebut, sistem dapat memperoleh gambaran mengenai kecenderungan preferensi musik pengguna pada konteks aktivitas yang telah dipilih sebelumnya.
Jawaban yang diberikan oleh pengguna pada setiap pertanyaan kemudian akan dikonversi oleh sistem menjadi nilai bobot preferensi yang digunakan dalam proses perhitungan metode EDAS. Dengan demikian, proses rekomendasi playlist yang dihasilkan dapat menyesuaikan karakteristik musik dengan preferensi pengguna secara lebih personal.
Pada bagian bawah halaman terdapat dua tombol navigasi, yaitu tombol "Kembali" yang berfungsi untuk kembali ke halaman input konteks aktivitas, serta tombol "Proses Rekomendasi" yang digunakan untuk melanjutkan ke tahap perhitungan rekomendasi playlist.

Pada bagian tengah halaman ditampilkan indikator loading yang menunjukkan bahwa sistem sedang memproses data yang diberikan oleh pengguna. Selain itu, halaman ini juga menampilkan beberapa tahapan proses yang dilakukan oleh sistem. Informasi tahapan proses ini bertujuan memberikan gambaran kepada pengguna mengenai langkah-langkah yang dilakukan sistem dalam menghasilkan rekomendasi playlist.

4. Halaman Hasil Rekomendasi Playlist
Halaman terakhir pada sistem merupakan halaman hasil rekomendasi playlist yang ditampilkan setelah sistem menyelesaikan proses perhitungan menggunakan metode EDAS. Pada halaman ini pengguna dapat melihat daftar lagu yang direkomendasikan oleh sistem beserta penjelasan mengenai alasan pemilihan lagu dalam playlist tersebut.
Pada bagian atas halaman ditampilkan judul "Playlist Rekomendasi untuk Aktivitas Anda" yang menunjukkan bahwa daftar lagu yang ditampilkan merupakan hasil rekomendasi sistem berdasarkan konteks aktivitas yang telah dimasukkan oleh pengguna pada tahap sebelumnya.
Di bawah judul halaman terdapat sebuah bagian informasi yang menampilkan ringkasan konteks aktivitas pengguna, yaitu aktivitas yang dipilih, mood pengguna, serta durasi aktivitas. Informasi ini ditampilkan kembali agar pengguna dapat memahami bahwa playlist yang dihasilkan merupakan hasil rekomendasi yang disesuaikan dengan konteks tersebut.
Bagian utama halaman menampilkan daftar lagu rekomendasi dalam bentuk tabel. Tabel tersebut terdiri dari beberapa kolom yaitu nomor urut lagu, judul lagu, nama artis, serta durasi lagu. Urutan lagu yang ditampilkan pada tabel merupakan hasil pemeringkatan yang diperoleh dari nilai Appraisal Score pada metode EDAS. Lagu dengan nilai kesesuaian yang lebih tinggi akan ditempatkan pada posisi yang lebih atas dalam daftar playlist.
Setelah daftar lagu ditampilkan, halaman ini juga menampilkan bagian penjelasan rekomendasi playlist yang dihasilkan oleh modul Natural Language Generation (NLG). Pada bagian ini sistem menyajikan narasi yang menjelaskan alasan pemilihan karakteristik musik dalam playlist yang direkomendasikan. Penjelasan tersebut disusun berdasarkan konteks aktivitas pengguna serta karakteristik umum dari lagu-lagu yang terpilih dalam playlist.
Selain narasi penjelasan, halaman ini juga menampilkan ringkasan karakteristik playlist dalam bentuk poin-poin singkat. Informasi tersebut menggambarkan karakteristik utama dari playlist yang dihasilkan, seperti tempo musik, tingkat energi lagu, serta dominasi unsur instrumental pada lagu-lagu yang direkomendasikan.
Pada bagian bawah halaman terdapat tombol "Buat Rekomendasi Baru" yang memungkinkan pengguna untuk kembali ke tahap awal sistem dan memasukkan konteks aktivitas yang berbeda apabila pengguna ingin mendapatkan rekomendasi playlist yang baru.
